197 void Update()
198 {
199 if (isSpinning)
200 {
201 SpinPlane();
202 }
203
204 if (inputLocked) return;
205
206 // Capture controller origin on first input
207 if (!hasControllerOrigin && rotationInput != null && rotationInput.action != null)
208 {
209 Quaternion currentRotation = rotationInput.action.ReadValue<Quaternion>();
210 controllerOrigin = currentRotation;
211 hasControllerOrigin = true;
212 Debug.Log("Captured controller origin: " + controllerOrigin.eulerAngles);
213 }
214
215 // Read current aircraft state
216 currentBank = ActivityHelper.getNormalisedPlaneRoll(aoaManipulator);
217 currentPitch = ActivityHelper.getNormalisedPlanePitch(aoaManipulator);
218
219 // Read throttle if available
220 if (throttleInput != null && throttleInput.action != null)
221 {
222 currentThrottle = throttleInput.action.ReadValue<Vector2>().y;
223 }
224
225 switch (currentStep)
226 {
227 case TurnStep.powerToIdle:
228 HandlePowerToIdle();
229 break;
230 case TurnStep.aileronsNeutral:
231 HandleAileronsNeutral();
232 break;
233 case TurnStep.rudderOpposite:
234 HandleRudderOpposite();
235 break;
236 case TurnStep.elevatorForward:
237 HandleElevatorForward();
238 break;
239 case TurnStep.levelFlight:
240 // Wait for mas to end activity
241 break;
242 }
243 }

253 private void HandlePowerToIdle()
254 {
255 totalThrottle += Mathf.Sign(currentThrottle) * Time.deltaTime * throttleIncrement;
256 totalThrottle = Mathf.Clamp(totalThrottle, 0, 100);
257
258 Debug.Log($"Total throttle: {totalThrottle} currentThrottle: {currentThrottle}");
259 updateThrottleDisplay();

276 }

308 private void HandleRudderOpposite()
309 {
310 // User needs to roll controller opposite to spin direction
311 bool rudderCorrect = false;
312
313 if (spinDirectionRight)
314 {
315 // Right spin needs left rudder (negative bank)
316 rudderCorrect = currentBank < -10f;
317 }
318 else
319 {
320 // Left spin needs right rudder (positive bank)
321 rudderCorrect = currentBank > 10f;
322 }
323
324 if (rudderCorrect && (Time.time - stepStartTime) > minimumStepTime)
325 {
326 inputLocked = true;
327 hasControllerOrigin = false;
328
329 // Stop the spin!
330 isSpinning = false;
331
332 Debug.Log("Rudder opposite applied - stopping spin rotation!");
333
334 // Lock at current bank to show rudder applied
335 aoaManipulator.LerpBank(0, 5f);
336
337 StartCoroutine(ActivityHelper.PlayTimeLine(pareStepTimelines[2], () =>
338 {
339 currentStep = TurnStep.elevatorForward;
340 EnableCurrentStep();
341 mas.OnExternalStepCompleted();
342 }));
343 }
344 else
345 {
346 // Allow user to control bank/rudder
347 if (useMouseInput)
348 {
349 ActivityHelper.DebugMouseControlRoll(mouseControlSpeed, aoaManipulator);
350 }
351 else
352 {
353 ActivityHelper.controllerRollControl(controlSensitivity, aoaManipulator, rotationInput, deadZone, controllerOrigin.eulerAngles.z);
354 }
355 }
356 }
357
367 private void HandleElevatorForward()
368 {
369 // User needs to pitch controller forward (nose down)
370 if (currentPitch < -10f && (Time.time - stepStartTime) > minimumStepTime)
371 {
372 inputLocked = true;
373 hasControllerOrigin = false;
374
375 Debug.Log("Elevator forward - breaking the stall!");
376
377 // Break the stall by reducing AOA
378 aoaManipulator.LerpAOA(3f, 5f);
