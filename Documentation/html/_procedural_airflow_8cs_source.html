213 void EmitParticles(float dt)
214 {
215 // calculate particles number of particles to spawn
216 emissionsNextFrame += emissionRate * dt;
217
218 // get integer value of particle spawn requirements and decrement counter
219 int particlesToEmit = Mathf.FloorToInt(emissionsNextFrame);
220 emissionsNextFrame -= particlesToEmit;
221
222 // Return if no particles to spawn
223 if (particlesToEmit <= 0) return;
224
225 // get normalised vector of the chord
226 Vector3 chordDir = (trailingEdge.position - leadingEdge.position).normalized;
227
228 for (int i = 0; i < particlesToEmit; i++)
229 {
230 // Set the spawn position of the particle
231 var emitParams = new ParticleSystem.EmitParams();
232 Vector3 spawnPos = leadingEdge.position - chordDir * forwardSpawnDistance;
233 emitParams.position = spawnPos;
234
235 emitParams.startLifetime = 5f;
236 emitParams.startSize = particleSize;
237
238 Vector3 toLeadingEdge = (leadingEdge.position - spawnPos).normalized;
239 emitParams.velocity = toLeadingEdge * Random.Range(topMinSpeed, topMaxSpeed);
240
241 // randomly assign particle as above or below the chord
242 // set random velocity between top and min speeds for update loop conditions
243 bool above = Random.value > 0.2f;
244 Vector3 velocity;
245 if (above)
246 {
247 Vector3 airflowDir = chordDir;
248 airflowDir.y *= airSeperationStrength;
249 airflowDir = airflowDir.normalized;
250 velocity = airflowDir * Random.Range(topMinSpeed, topMaxSpeed);
251 sideFlags.Add(true);
252 }
253 else
254 {
255 velocity = chordDir * Random.Range(bottomMinSpeed, bottomMaxSpeed);
256 sideFlags.Add(false);
257 }
258
259 // Store the per particle velocities
260 velocities.Add(velocity);
261
262 // add particles to local buffer and particle system
263 ps.Emit(emitParams, 1);
264 }

278 void UpdateParticles(float dt)
279 {
280 int count = ps.GetParticles(buffer);
281
282 // the local position of leading and trailing edges
283 Vector3 localLeading = transform.InverseTransformPoint(leadingEdge.position);
284 Vector3 localTrailing = transform.InverseTransformPoint(trailingEdge.position);
285
286 // calculate chord direction in world space
287 Vector3 chordDir = (trailingEdge.position - leadingEdge.position).normalized;
288 // calculate the chord length in local space
289 float chordLengthLocal = Vector3.Distance(localTrailing, localLeading);
290
291 // determine the chord orientation in local space. sign determines if teh trailing edge is at +x or -x in local space
292 float sign = Mathf.Sign(localTrailing.x - localLeading.x);
293
294 // calculate poiints for particle direction changes
295 float deflectionStart = chordLengthLocal * .10f; // 20% of the chord length from leading edge
296 float trailDistance = chordLengthLocal * .50f; // 50% of the chord length past trailing edge
297
298 // Loop over all active particles
299 for (int i = 0; i < count; i++)
300 {
301 // get particle world position and convert to local position
302 Vector3 particleWorld = buffer[i].position;
303 Vector3 particleLocal = transform.InverseTransformPoint(particleWorld);
304
305 // Determines a particles position along the chord / displacement from leading edge
306 // sign ensures distances along the chord are positive
307 float alongChord = (particleLocal.x - localLeading.x) * sign;
308
309 // check is particles are over trail distance
310 if (alongChord > chordLengthLocal + trailDistance)
311 {
312 // Remove particle by swapping with last and reducing count
313 int lastIdx = count - 1;
314 if (i != lastIdx)
315 {
316 buffer[i] = buffer[lastIdx];
317 velocities[i] = velocities[lastIdx];
318 sideFlags[i] = sideFlags[lastIdx];
319 }
320 count--;
321 velocities.RemoveAt(lastIdx);
322 sideFlags.RemoveAt(lastIdx);
323 continue;
324 }
325
326 // if particle is at or past deflection point deflect based on side flags
327 if (alongChord >= -deflectionStart && alongChord <= 0f)
328 {
329 // Deflect relative to planes local up
330 Vector3 offsetTargetLocal;
331 if (sideFlags[i])
332 {
333 // Particle is above the wing, deflect upwards
334 offsetTargetLocal = new Vector3(
335 localLeading.x + deflectionStart * sign,
336 localLeading.y + deflectionStart,
337 particleLocal.z
338 );
339 }
340 else
341 {
342 // Particle is below the wing deflect downwards
343 offsetTargetLocal = new Vector3(
344 localLeading.x + deflectionStart * sign,
345 localLeading.y - deflectionStart,
346 particleLocal.z
347 );
348 }
349
350 // convert offset target to world coorinates
351 Vector3 offsetTargetWorld = transform.TransformPoint(offsetTargetLocal);
352 // get direction in world coordinates to target
353 Vector3 targetDir = (offsetTargetWorld - particleWorld).normalized;
354
355 float speed = velocities[i].magnitude;
356 buffer[i].velocity = targetDir * speed;
357 }
358
359 // if particle is past the deflection target (leading edge) set velocity to stored velocities value
360 else if (alongChord > deflectionStart && alongChord <= chordLengthLocal)
361 {
362 buffer[i].velocity = velocities[i];
363 }
364 // If particle is past trailing edge remove vertical velocity component
365 else if (alongChord > chordLengthLocal && alongChord <= chordLengthLocal + trailDistance)
366 {
367 float horizontalSpeed = velocities[i].magnitude;
368 // Remove y component from chordDir and normalize
369 Vector3 chordDirNoY = new Vector3(chordDir.x, chordDir.y * trailDownForcePercent, chordDir.z).normalized;
370 buffer[i].velocity = chordDirNoY * horizontalSpeed;
371 }
372 else
373 {
374 int n = 1;
375 }
376
377 // Apply velocity update to position
378 buffer[i].position += buffer[i].velocity * dt;
379 }
380
381 // Hand updated particles back to the system for rendering
382 ps.SetParticles(buffer, count);
383 }

487 private float InterpolateForceValue(float aoa, float[] forceValues)
488 {
489 // Handle values below minimum stage
490 if (aoa <= aoaStages[0]) return forceValues[0];
491
492 // Handle values above maximum stage
493 if (aoa >= aoaStages[aoaStages.Length - 1])
494 return forceValues[forceValues.Length - 1];
495
496 // Find the appropriate stage interval and interpolate
497 for (int i = 0; i < aoaStages.Length - 1; i++)
498 {
499 if (aoa <= aoaStages[i + 1])
500 {
501 // Calculate interpolation factor 0 = start, 1 = end, 0.5 = halfway
502 float t = (aoa - aoaStages[i]) / (aoaStages[i + 1] - aoaStages[i]);
503
504 // Linearly interpolate between lower and upper force values
505 return Mathf.Lerp(forceValues[i], forceValues[i + 1], t);
506 }
507 }
508 // Fallback to maximum value
509 return forceValues[forceValues.Length - 1];
510 }

524 void SetArrow(GameObject arrow, Vector3 basePos, Vector3 dir, float length)
525 {
526 float width = 0.1f; // arrow width
527 float height = 0.1f; // arrow height
528 float headSize = .2f; // Arrow head size
529
530 //Debug.Log(arrow.name + " " + length);
531
532 // Scale the arrow
533 arrow.transform.localScale = new Vector3(width, height, length);
534
535 // Make the arrow point along the force direction
536 Quaternion lookRotation = Quaternion.LookRotation(dir, Vector3.up);
537 // Create a quaternion consisting only of the planes y rotation
538 Quaternion yRotation = Quaternion.Euler(0f, transform.eulerAngles.y, 0f);
539 // apply the planes y rotation then rotate the arrow to point along force direction
540 arrow.transform.rotation = yRotation * lookRotation;
541
542 // Set the arrows position so the base is along the anchor point.
543 arrow.transform.position = basePos + arrow.transform.forward * (length * 0.5f);
544
545 // Get the arrow head
546 Transform headTransform = arrow.transform.GetChild(0);
547 // Set the head position to the shafts forward face
548 headTransform.localPosition = new Vector3(0, 0, 0.5f);
549 // compensate for the shafts scale.
550 // localScale * parentScale = worldScale therefore localScale = worldScale / parentScale
551 headTransform.localScale = new Vector3(headSize / width, headSize / height, headSize / length);
552 }
