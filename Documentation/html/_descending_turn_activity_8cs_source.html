208 public void HandleBankAndPitch()

211 if (!hasControllerOrigin)

218 float relativeZ = Mathf.DeltaAngle(0f, relativeRotation.eulerAngles.z);
219 float relativeX = Mathf.DeltaAngle(0f, relativeRotation.eulerAngles.x);
220
221 float planeRoll = ActivityHelper.getNormalisedPlaneRoll(aoaManipulator);
222 float planePitch = ActivityHelper.getNormalisedPlanePitch(aoaManipulator);
223
224 bool rollAchieved = ActivityHelper.checkExactRotationTargetAchieved(planeRoll, 30f, stepAdvanceTolerance);
225 bool pitchAchieved = ActivityHelper.checkExactRotationTargetAchieved(planePitch, -5f, stepAdvanceTolerance);
226
227 Debug.Log("rollAchieved: " + rollAchieved + " pitchAchieved: " + pitchAchieved);
228 if (rollAchieved)

230 rollInputLocked = true;

233 if (pitchAchieved)
234 {
235 // Here, lerp causes issue. fix tomorrow
236 pitchInputLocked = true;
237 aoaManipulator.LerpAOA(-5f);
238 }
239
240 if (rollInputLocked && pitchInputLocked)
241 {
242 currentStep = TurnStep.PlayingTimeline;
243 hasControllerOrigin = false;
244 Debug.Log("Both roll and pitch achieved, starting coroutine...");
245 StartCoroutine(ActivityHelper.PlayTimeLine(postBankAndPitchDirector, () =>
246 {
247 Debug.Log("Timeline complete, advancing to PitchUp");

288 public void HandlePitch()
289 {
290 if (pitchInputLocked) return;
291
292 Quaternion currentRotation = rotationInput.action.ReadValue<Quaternion>();
293 if (!hasControllerOrigin)
294 {
295 controllerOrigin = currentRotation;
296 hasControllerOrigin = true;
297 }
298
299 float currentPitch = ActivityHelper.getNormalisedPlanePitch(aoaManipulator);
300
301 if (ActivityHelper.checkExactRotationTargetAchieved(currentPitch, 5f, stepAdvanceTolerance))
302 {
303 Debug.Log($"Pitch target achieved: {currentPitch}");
304 pitchInputLocked = true;
305 hasControllerOrigin = false;
306 aoaManipulator.LerpAOA(5f, 0.5f);
307 StartCoroutine(ActivityHelper.PlayTimeLine(postPitchUpDirector, () =>
308 {
309 pitchInputLocked = false;
310 currentStep = TurnStep.Stall;

312 mas.OnExternalStepCompleted();
313 }));

322 {
323 ActivityHelper.controllerPitchControl(bankSensitivity, aoaManipulator, rotationInput, deadZone, controllerOrigin.eulerAngles.x);
