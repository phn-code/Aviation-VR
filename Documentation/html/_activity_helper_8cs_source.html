69 public static PlayableDirector[] FindPlayableDirector(TimelineAsset[] assets)
70 {
71 // Check for null or empty array
72 if (assets == null || assets.Length == 0)
73 {
74 Debug.LogWarning("TimeLineAsset array is NULL or empty: Cannot find PlayableDirectors");
75 return null;
76 }
77
78 // Get all PlayableDirectors in the scene
79 PlayableDirector[] allDirectors = Object.FindObjectsOfType<PlayableDirector>();
80 // Create an array to hold found Directors
81 PlayableDirector[] foundDirectors = new PlayableDirector[assets.Length];
82
83 // Search for Directors for each asset
84 for (int i = 0; i < assets.Length; i++)
85 {
86 // Get the next asset
87 TimelineAsset asset = assets[i];
88 foundDirectors[i] = null;
89
90 // Check for null asset
91 if (asset != null)
92 {
93 // Search for the Director that uses this asset
94 for (int j = 0; j < allDirectors.Length; j++)
95 {
96 PlayableDirector director = allDirectors[j];
97 if (director.playableAsset == asset)
98 {
99 foundDirectors[i] = director;
100 break;
101 }
102 }
103 // If still null
104 if (foundDirectors[i] == null)
105 {
106 Debug.LogWarning("No PlayableDirector found for timeline asset: " + asset.name);
107 }
108 }
109 else
110 {
111 Debug.LogWarning("TimelineAsset at index " + i + " is NULL: Cannot find PlayableDirector");
112 }
113 }
114 return foundDirectors;
115 }

354 public static void controllerRollControl(float speed, AxisRotationController manip, InputActionReference input, float deadZone, float initial = 0, Vector2 vector = default)
355 {
356 if (vector == default)
357 {
358 vector.x = -70f;
359 vector.y = 70f;
360 }
361
362 float planeX = Mathf.DeltaAngle(0, manip.getRotation().eulerAngles.x);
363 planeX = clampRange(planeX, vector);
364 manip.setRoll(planeX);
365
366 float zRotation = normalizedControllerZRotation(input, initial);
367 if (Mathf.Abs(zRotation) > deadZone)
368 {
369 zRotation = getRotationSpeed(zRotation, speed, deadZone);
370 manip.IncrementBank(zRotation * Time.deltaTime);

433 public static float getRotationSpeed(float rotation, float speed, float deadZone)
434 {

524 public static IEnumerator PlayTimeLine(PlayableDirector director, System.Action onComplete, float fallBackWait = 1f)
525 {
526 if (director != null && director.playableAsset != null)
527 {
528 director.Play();
529
530 bool isComplete = false;
531 // Subscribe to the stopped event
532 // Lambda called when timeline ends setting isComplete to true
533 director.stopped += (d) => { isComplete = true; };
534
535 // Wait until timeline is complete
536 while (!isComplete)
537 {
538 yield return null;
539 }
540 // Call lambda fuction passed in to signal activity complete
541 onComplete.Invoke();
542 }
543 else
544 {
545 Debug.Log("Yeilding as fallback for unassigned PlayableDirector (Timeline)");
546 yield return new WaitForSeconds(fallBackWait);
547 onComplete.Invoke();
548 }
