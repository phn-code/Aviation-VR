137 void Update()
138 {
139 // Linear increase and decrease of propeller speed when toggled on and off respectively
140 if (PropellerEnabled && Helice != null) // Rotate the propeller if it is toggled on
141 {
142 currentPropellerSpeed = Mathf.MoveTowards(currentPropellerSpeed, PropellerRotationSpeed, propellerAcceleration * Time.deltaTime);
143
144 Helice.transform.Rotate(Vector3.right, -currentPropellerSpeed * Time.deltaTime);
145 }
146 else
147 {
148 currentPropellerSpeed = Mathf.MoveTowards(currentPropellerSpeed, 0f, propellerAcceleration * Time.deltaTime);
149 Helice.transform.Rotate(Vector3.right, -currentPropellerSpeed * Time.deltaTime);
150 }
151
152 if (GhostTrailEnabled != lastTrailState) // Every frame, turn on the ghost trail if enabled by checking prev state
153 {
154 if (GhostTrailEnabled)
155 GhostTrail.Play();
156 else
157 GhostTrail.Stop(true, ParticleSystemStopBehavior.StopEmitting);
158
159 lastTrailState = GhostTrailEnabled;
160 }
161
162 // This is all deprecated!!! Doesn't need to be here anymore, but it is disabled in practice already.
163 for (int i = 0; i < AxialScrollingPlanes.Count; i++) // For every axial plane (should only be 3)
164 {
165 var axis = AxialScrollingPlanes[i];
166 if (axis == null) continue;
167
168 var renderers = axis.GetComponentsInChildren<Renderer>(true);
169 Vector2 offset = Vector2.zero; // Offset will be what the texture uses to scroll
170
171 switch (i) // Each plane will read different position values to determine their axis' offset
172 {
173 case 0: // X plane offset - "back", relies on vehicle's Y and Z positions
174 offset = new Vector2(transform.position.y, transform.position.z) * AxialScrollSpeed;
175 //Debug.Log("Changing stuff!");
176 break;
177 case 1: // Y plane offset - "left", relies on vehicle's X and Z positions
178 offset = new Vector2(transform.position.x, transform.position.z) * AxialScrollSpeed;
179 //Debug.Log("Changing stuff!");
180 break;
181 case 2: // Z plane offset - "up", relies on vehicle's X and Y positions
182 offset = new Vector2(transform.position.y, transform.position.x) * AxialScrollSpeed;
183 //Debug.Log("Changing stuff!");
184 break;
185 }
186
187 foreach (var renderer in renderers) // Apply the offset
188 {
189 renderer.material.SetTextureOffset("_BaseMap", offset);
190 }
191 }
