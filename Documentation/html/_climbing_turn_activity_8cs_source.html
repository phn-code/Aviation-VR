204 public void HandleBanking()
205 {
206 if (rollInputLocked) return;
207
208 Quaternion currentRotation = rotationInput.action.ReadValue<Quaternion>();
209
210 if (!hasControllerOrigin)
211 {
212 controllerOrigin = currentRotation;
213 hasControllerOrigin = true;
214 Debug.Log("Captured controller origin (Bank): " + controllerOrigin.eulerAngles);
215 }
216
217 currentBank = ActivityHelper.getNormalisedPlaneRoll(aoaManipulator);
218
219 if (ActivityHelper.checkExactRotationTargetAchieved(currentBank, 15f, stepAdvanceTolerance))
220 {
221 hasControllerOrigin = false;
222 rollInputLocked = true;
223 aoaManipulator.LerpBank(15f);
224 StartCoroutine(ActivityHelper.PlayTimeLine(postBankDirector, () =>
225 {
226 pitchInputLocked = false;
227 currentStep = TurnStep.Stall;
228 // TODO uncomment after testing
229 mas.OnExternalStepCompleted();
230 }));
231 }
232 else
233 {
234 if (useMouseClicks)
235 {
236 ActivityHelper.DebugMouseControlRoll(mouseRollSpeed, aoaManipulator);
237 }
238 else
239 {
240 ActivityHelper.controllerRollControl(bankSensitivity, aoaManipulator, rotationInput, deadZone, controllerOrigin.eulerAngles.z);
241 }
242 }
243 }

255 public void HandleStall()
